{{/* Builds a BreadcrumbList JSON-LD node. Context: page. The list always starts with the home page and follows Hugo's .Ancestors chain up to the current page. Returns a BreadcrumbList dict, or empty dict if the page is the home page (single-item breadcrumbs are meaningless). */}} {{- $permalink := .Permalink | strings.TrimRight "/" -}} {{- $items := slice -}} {{- $pos := 1 -}} {{- /* Home */ -}} {{- $items = $items | append (dict "@type" "ListItem" "position" $pos "name" site.Title "item" site.BaseURL ) -}} {{- $pos = add $pos 1 -}} {{- /* Ancestors in ascending order (root → parent) */ -}} {{- range .Ancestors.Reverse -}} {{- if ne .Kind "home" -}} {{- $items = $items | append (dict "@type" "ListItem" "position" $pos "name" .Title "item" .Permalink ) -}} {{- $pos = add $pos 1 -}} {{- end -}} {{- end -}} {{- /* Current page (only if not home) */ -}} {{- if ne .Kind "home" -}} {{- $items = $items | append (dict "@type" "ListItem" "position" $pos "name" .Title "item" .Permalink ) -}} {{- end -}} {{- /* A single-item list (home only) is not useful */ -}} {{- if le (len $items) 1 -}} {{- return dict -}} {{- end -}} {{- return (dict "@type" "BreadcrumbList" "@id" (printf "%s#breadcrumb" $permalink) "itemListElement" $items ) -}}
