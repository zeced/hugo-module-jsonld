{{/* Builds Place JSON-LD nodes from data/jsonld/places.json (root array). Outputs JSON array. Callers: partial "..." . | transform.Unmarshal (then range) Caching: hugo.Store key "jsonld:places" (JSON string). */}} {{- $cachedJSON := hugo.Store.Get "jsonld:places" | default "" -}} {{- if $cachedJSON -}} {{- $cachedJSON -}} {{- else -}} {{- $placesData := index hugo.Data "jsonld" | default dict -}} {{- $places := index $placesData "places" | default slice -}} {{- $siteURL := site.BaseURL | strings.TrimRight "/" -}} {{- $nodes := slice -}} {{- range $places -}} {{- if not .slug -}} {{- warnf "[jsonld] Place entry %q has no slug — skipping." .name -}} {{- continue -}} {{- end -}} {{- $address := dict "@type" "PostalAddress" "streetAddress" .address.streetAddress "addressLocality" .address.addressLocality "postalCode" .address.postalCode "addressCountry" .address.addressCountry -}} {{- $node := dict "@type" "Place" "@id" (printf "%s#place-%s" $siteURL .slug) "name" .name "address" $address -}} {{- if .openingHours -}} {{- $specs := slice -}} {{- range .openingHours -}} {{- $specs = $specs | append (dict "@type" "OpeningHoursSpecification" "dayOfWeek" .dayOfWeek "opens" .opens "closes" .closes ) -}} {{- end -}} {{- $node = merge $node (dict "openingHoursSpecification" $specs) -}} {{- end -}} {{- with .telephone -}}{{- $node = merge $node (dict "telephone" .) -}}{{- end -}} {{- with .url -}}{{- $node = merge $node (dict "url" .) -}}{{- end -}} {{- $nodes = $nodes | append $node -}} {{- end -}} {{- $json := $nodes | jsonify -}} {{- hugo.Store.Set "jsonld:places" $json -}} {{- $json -}} {{- end -}}
