{{/* Builds a FAQPage JSON-LD node. Context: page. FAQ sources (combined, deduplicated by question text): 1. Global FAQs — data/jsonld/faqs.json Included when the page's .Type is listed in params.jsonld.faqSections. Each entry: { question, answer, active (bool), pinned (bool) } - active: false → always skipped - pinned: true → also included on pages that have page-level FAQs, even if the section is not in faqSections 2. Page-level FAQs — params.jsonld.faqs (frontmatter) Always included when present. Each entry: { question, answer } Config (params.jsonld): faqSections []string — .Type values where global FAQs are injected Returns a FAQPage dict, or empty dict if no FAQ items are found. */}} {{- $jsonld := .Params.jsonld | default dict -}} {{- $cfg := site.Params.jsonld | default dict -}} {{- $faqSections := $cfg.faqSections | default slice -}} {{- $pageFaqs := $jsonld.faqs -}} {{- $inSection := in $faqSections .Type -}} {{- /* Nothing to do */ -}} {{- if not (or $pageFaqs $inSection) -}} {{- return dict -}} {{- end -}} {{- $items := slice -}} {{- $seen := slice -}} {{- /* deduplicate by question text */ -}} {{- /* ── Global FAQs ── */ -}} {{- $faqData := index site.Data "jsonld" | default dict -}} {{- $globalFaqs := index $faqData "faqs" | default slice -}} {{- range $globalFaqs -}} {{- if not .active -}}{{- continue -}}{{- end -}} {{- $include := false -}} {{- if $inSection -}} {{- $include = true -}} {{- else if and $pageFaqs .pinned -}} {{- /* Pinned globals surface on any page that has its own FAQs */ -}} {{- $include = true -}} {{- end -}} {{- if and $include (not (in $seen .question)) -}} {{- $items = $items | append (dict "@type" "Question" "name" .question "acceptedAnswer" (dict "@type" "Answer" "text" .answer) ) -}} {{- $seen = $seen | append .question -}} {{- end -}} {{- end -}} {{- /* ── Page-level FAQs ── */ -}} {{- range $pageFaqs -}} {{- if not (in $seen .question) -}} {{- $items = $items | append (dict "@type" "Question" "name" .question "acceptedAnswer" (dict "@type" "Answer" "text" .answer) ) -}} {{- $seen = $seen | append .question -}} {{- end -}} {{- end -}} {{- if not $items -}} {{- return dict -}} {{- end -}} {{- $permalink := .Permalink | strings.TrimRight "/" -}} {{- return (dict "@type" "FAQPage" "@id" (printf "%s#faqpage" $permalink) "mainEntity" $items "isPartOf" (dict "@id" (printf "%s#webpage" $permalink)) ) -}}
