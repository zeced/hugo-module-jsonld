{{/* Builds a FAQPage JSON-LD node. Outputs JSON (empty object {} if no FAQ items found). Callers: partial "..." . | transform.Unmarshal FAQ sources: 1. Global (data/jsonld/faqs.json) — when .Type is in params.jsonld.faqSections pinned:true entries also appear on pages that have their own jsonld.faqs 2. Page-level (params.jsonld.faqs) — always when present */}} {{- $jsonld := .Params.jsonld | default dict -}} {{- $cfg := site.Params.jsonld | default dict -}} {{- $faqSections := $cfg.faqSections | default slice -}} {{- $pageFaqs := $jsonld.faqs -}} {{- $inSection := in $faqSections .Type -}} {{- $node := dict -}} {{- $items := slice -}} {{- $seen := slice -}} {{- if or $pageFaqs $inSection -}} {{- $faqData := index hugo.Data "jsonld" | default dict -}} {{- $globalFaqs := index $faqData "faqs" | default slice -}} {{- range $globalFaqs -}} {{- if not .active -}}{{- continue -}}{{- end -}} {{- $include := or $inSection (and $pageFaqs .pinned) -}} {{- if and $include (not (in $seen .question)) -}} {{- $items = $items | append (dict "@type" "Question" "name" .question "acceptedAnswer" (dict "@type" "Answer" "text" .answer) ) -}} {{- $seen = $seen | append .question -}} {{- end -}} {{- end -}} {{- range $pageFaqs -}} {{- if not (in $seen .question) -}} {{- $items = $items | append (dict "@type" "Question" "name" .question "acceptedAnswer" (dict "@type" "Answer" "text" .answer) ) -}} {{- $seen = $seen | append .question -}} {{- end -}} {{- end -}} {{- if $items -}} {{- $permalink := .Permalink | strings.TrimRight "/" -}} {{- $node = dict "@type" "FAQPage" "@id" (printf "%s#faqpage" $permalink) "mainEntity" $items "isPartOf" (dict "@id" (printf "%s#webpage" $permalink)) -}} {{- end -}} {{- end -}} {{- $node | jsonify -}}
