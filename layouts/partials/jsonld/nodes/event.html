{{/* Builds an Event JSON-LD node. Outputs JSON (empty object {} if startDate missing/invalid). Callers: partial "..." . | transform.Unmarshal Frontmatter (params.jsonld) — startDate is required. See README for full schema. */}} {{- $siteURL := site.BaseURL | strings.TrimRight "/" -}} {{- $orgID := printf "%s#organization" $siteURL -}} {{- $permalink := .Permalink | strings.TrimRight "/" -}} {{- $jsonld := .Params.jsonld | default dict -}} {{- $name := $jsonld.name | default .Title -}} {{- $description := $jsonld.description | default .Description | default .Summary -}} {{- $startDate := partial "jsonld/helpers/validate-date.html" (dict "date" $jsonld.startDate "page" .) -}} {{- $endDate := partial "jsonld/helpers/validate-date.html" (dict "date" $jsonld.endDate "page" .) -}} {{- $node := dict -}} {{- if $startDate -}} {{- /* Location: frontmatter > primary place */ -}} {{- $location := dict -}} {{- $jLocation := $jsonld.location | default dict -}} {{- with $jLocation.name -}} {{- $loc := dict "@type" "Place" "name" . -}} {{- with $jLocation.address -}} {{- $loc = merge $loc (dict "address" (dict "@type" "PostalAddress" "streetAddress" .streetAddress "addressLocality" .addressLocality "postalCode" .postalCode "addressCountry" .addressCountry )) -}} {{- end -}} {{- $location = $loc -}} {{- else -}} {{- $placesData := index hugo.Data "jsonld" | default dict -}} {{- $places := index $placesData "places" | default slice -}} {{- with index (where $places "administrative" true) 0 -}} {{- $location = dict "@id" (printf "%s#place-%s" $siteURL .slug) -}} {{- end -}} {{- end -}} {{- /* Organizer: frontmatter > Organization */ -}} {{- $organizer := dict "@id" $orgID -}} {{- $jOrganizer := $jsonld.organizer | default dict -}} {{- with $jOrganizer.name -}} {{- $organizer = dict "@type" "Organization" "name" . -}} {{- with $jOrganizer.url -}}{{- $organizer = merge $organizer (dict "url" .) -}}{{- end -}} {{- end -}} {{- $image := partial "jsonld/helpers/resolve-image.html" (dict "page" .) -}} {{- $node = dict "@type" "Event" "@id" (printf "%s#event" $permalink) "name" $name "startDate" $startDate "url" .Permalink "organizer" $organizer "eventAttendanceMode" "https://schema.org/OfflineEventAttendanceMode" "eventStatus" "https://schema.org/EventScheduled" -}} {{- with $endDate -}}{{- $node = merge $node (dict "endDate" .) -}}{{- end -}} {{- with $description -}}{{- $node = merge $node (dict "description" .) -}}{{- end -}} {{- with $image -}}{{- $node = merge $node (dict "image" .) -}}{{- end -}} {{- with $location -}}{{- $node = merge $node (dict "location" .) -}}{{- end -}} {{- $jOffers := $jsonld.offers | default slice -}} {{- if $jOffers -}} {{- $offers := slice -}} {{- range $jOffers -}} {{- $avail := printf "https://schema.org/%s" (.availability | default "InStock") -}} {{- $offer := dict "@type" "Offer" "price" (string .price) "priceCurrency" (.priceCurrency | default "EUR") "availability" $avail -}} {{- with .name -}}{{- $offer = merge $offer (dict "name" .) -}}{{- end -}} {{- with .url -}}{{- $offer = merge $offer (dict "url" .) -}}{{- end -}} {{- $offers = $offers | append $offer -}} {{- end -}} {{- if eq (len $offers) 1 -}} {{- $node = merge $node (dict "offers" (index $offers 0)) -}} {{- else -}} {{- $node = merge $node (dict "offers" $offers) -}} {{- end -}} {{- end -}} {{- else -}} {{- warnf "[jsonld] %q: Event requires jsonld.startDate in frontmatter." .RelPermalink -}} {{- end -}} {{- $node | jsonify -}}
