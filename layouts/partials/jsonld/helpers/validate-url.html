{{/* Validates that a URL is absolute. Input dict: url string — URL to validate page Page — page context for warning messages Returns the URL if valid, empty string otherwise (with a warnf). */}} {{- $url := .url | default "" -}} {{- $page := .page -}} {{- if not $url -}}{{- return "" -}}{{- end -}} {{- $valid := or (strings.HasPrefix $url "https://") (strings.HasPrefix $url "http://") -}} {{- if not $valid -}} {{- warnf "[jsonld] %q: URL %q is not absolute — use an absolute URL starting with https://" $page.RelPermalink $url -}} {{- return "" -}} {{- end -}} {{- return $url -}}
