{{/* Validates that a URL is absolute. Input dict: url string — URL to validate page Page — page context for warning messages Outputs the URL if valid, nothing (empty string) otherwise. Callers: $u := partial "jsonld/helpers/validate-url.html" (dict ...) */}} {{- $url := .url | default "" -}} {{- $page := .page -}} {{- if $url -}} {{- if or (strings.HasPrefix $url "https://") (strings.HasPrefix $url "http://") -}} {{- $url -}} {{- else -}} {{- warnf "[jsonld] %q: URL %q is not absolute — use an absolute URL starting with https://" $page.RelPermalink $url -}} {{- end -}} {{- end -}}
