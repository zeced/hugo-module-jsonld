{{/* Validates an ISO 8601 date string. Input dict: date string — date to validate page Page — page context for warning messages Outputs the date string if valid, nothing (empty string) otherwise. Callers: $d := partial "jsonld/helpers/validate-date.html" (dict ...) */}} {{- $date := .date | default "" -}} {{- $page := .page -}} {{- if $date -}} {{- $valid := and (ge (len $date) 10) (eq (substr $date 4 1) "-") (eq (substr $date 7 1) "-") -}} {{- if $valid -}} {{- $date -}} {{- else -}} {{- warnf "[jsonld] %q: invalid date %q — expected ISO 8601, e.g. 2024-06-15T10:00:00+02:00" $page.RelPermalink $date -}} {{- end -}} {{- end -}}
